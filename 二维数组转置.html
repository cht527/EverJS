function arrayTranspose(array){
  const rows=array.length;
  const columns=array[0].length;
  const target=[]
  for(let i=0;i < rows;i++){
    target[i]=[];
    for(var j=0;j < array[i].length;j++){
        target[i][j]=array[j][i];
    }
  }
  return target
}
