// 生成

const produceMatrix = (m,n,isNoRepeat) => {
  const _matrix = [];let tempMax=0;
  for (let i = 0; i < m; i++) {
      const temp=[];
      for(let j=0;j < n;j++){
          temp.push(i+j+tempMax)
      }
      if(isNoRepeat){
       tempMax+=(n-1);
      }
      _matrix.push(temp)
  }
  return _matrix
}

function arrayTranspose(array){

  const target=[]
